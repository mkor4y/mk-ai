    def safe_html_message(self, message):
        """Markdown mesajını güvenli HTML'e çevir"""
        if not message:
            return ""
        
        # Markdown'ı HTML'e çevir
        html_message = message.replace('**', '<b>').replace('**', '</b>')
        html_message = html_message.replace('*', '<i>').replace('*', '</i>')
        html_message = html_message.replace('`', '<code>').replace('`', '</code>')
        html_message = html_message.replace('\n', '<br>')
        
        # Özel karakterleri escape et
        html_message = html_message.replace('&', '&amp;')
        html_message = html_message.replace('<', '&lt;').replace('>', '&gt;')
        
        return html_message

    async def get_crypto_analysis_data(self, coin_id, days=30):
        """Kripto için veri, teknik analiz, sinyal ve market_data döndürür"""
        df = self.crypto_analyzer.get_crypto_data(coin_id, days=days)
        if df is None:
            return None, None, None, "❌ {0} için veri bulunamadı.".format(coin_id)
        df = self.crypto_analyzer.calculate_technical_indicators(df)
        if df is None:
            return None, None, None, "❌ {0} için teknik analiz yapılamadı.".format(coin_id)
        signals = self.crypto_analyzer.generate_trading_signals(df)
        if not signals:
            return None, None, None, "❌ {0} için sinyal üretilemedi.".format(coin_id)
        market_data = self.crypto_analyzer.get_market_summary(coin_id)
        if market_data is None:
            return None, None, None, "❌ {0} için piyasa verisi bulunamadı.".format(coin_id)
        return df, signals, market_data, None

    async def analyze_coin(self, update: Update, context: ContextTypes.DEFAULT_TYPE):
        try:
            if context.args:
                coin_id = context.args[0].lower()
            else:
                await update.message.reply_text("❌ Lütfen bir kripto para adı girin.\nÖrnek: /analyze bitcoin")
                return
            status_message = await update.message.reply_text("🔍 Analiz yapılıyor...")
            df, signals, market_data, error = await self.get_crypto_analysis_data(coin_id)
            if error:
                await status_message.edit_text(error)
                return
            analysis_message = self.crypto_analyzer.format_analysis_message(coin_id, df, signals, market_data)
            ai_analysis = self.chatgpt.get_ai_analysis(market_data['name'], df, market_data, signals)
            full_message = analysis_message + "\n\n🤖 **AI Analizi:**\n" + ai_analysis
            await status_message.edit_text(full_message, parse_mode='HTML')
        except Exception as e:
            logger.error(f"Analiz hatası: {e}")
            await update.message.reply_text("❌ Analiz sırasında hata oluştu.")

    async def market_summary(self, update: Update, context: ContextTypes.DEFAULT_TYPE):
        """Piyasa özeti"""
        try:
            status_message = await update.message.reply_text("📊 Piyasa özeti hazırlanıyor...")
            
            # Desteklenen coinlerin özetini al
            summary_data = []
            for coin_id in self.config.SUPPORTED_COINS[:5]:  # İlk 5 coin
                market_data = self.crypto_analyzer.get_market_summary(coin_id)
                if market_data:
                    summary_data.append(market_data)
            
            if not summary_data:
                await status_message.edit_text("❌ Piyasa verisi alınamadı.")
                return
            
            # Özet mesajını oluştur
            summary_message = "📊 **Piyasa Özeti**\n\n"
            
            for data in summary_data:
                change_emoji = "📈" if data['price_change_24h'] > 0 else "📉"
                summary_message += f"{change_emoji} **{data['symbol']}:** ${data['current_price']:,.2f} ({data['price_change_24h']:+.2f}%)\n"
            
            # AI piyasa duyarlılığı
            coin_names = [data['name'] for data in summary_data]
            sentiment = self.chatgpt.get_market_sentiment(coin_names)
            
            summary_message += f"\n🤖 **AI Piyasa Değerlendirmesi:**\n{sentiment}"
            
            await status_message.edit_text(summary_message, parse_mode='Markdown')
            
        except Exception as e:
            logger.error(f"Piyasa özeti hatası: {e}")
            await update.message.reply_text("❌ Piyasa özeti oluşturulurken hata oluştu.")

    async def button_callback(self, update: Update, context: ContextTypes.DEFAULT_TYPE):
        """Buton callback'leri"""
        query = update.callback_query
        await query.answer()
        
        if query.data.startswith("analyze_"):
            coin_id = query.data.replace("analyze_", "")
            # Analiz işlemini başlat
            context.args = [coin_id]
            # Callback query için özel update objesi oluştur
            fake_update = type('Update', (), {

    async def error_handler(self, update: Update, context: ContextTypes.DEFAULT_TYPE):
        """Hata yönetimi"""
        logger.error(f"Bot hatası: {context.error}")
        if update and update.effective_message:
            await update.effective_message.reply_text("❌ Bir hata oluştu. Lütfen daha sonra tekrar deneyin.")
    
    def run(self):
        """Botu çalıştır"""
        # Bot uygulamasını oluştur
        application = Application.builder().token(self.config.TELEGRAM_TOKEN).build()
        
        # Komut handler'larını ekle
        application.add_handler(CommandHandler("start", self.start))
        
        # Kripto para komutları
        application.add_handler(CommandHandler("analyze", self.analyze_coin))
        application.add_handler(CommandHandler("price", self.get_price))
        application.add_handler(CommandHandler("signals", self.get_signals))
        application.add_handler(CommandHandler("ai_analysis", self.ai_analysis))
        application.add_handler(CommandHandler("market_summary", self.market_summary))
        
        # BIST komutları
        application.add_handler(CommandHandler("analyze_bist", self.analyze_bist))
        application.add_handler(CommandHandler("price_bist", self.get_price_bist))
        application.add_handler(CommandHandler("signals_bist", self.get_signals_bist))
        application.add_handler(CommandHandler("ai_analysis_bist", self.ai_analysis_bist))
        application.add_handler(CommandHandler("market_summary_bist", self.market_summary_bist))
        
        # Genel komutlar
        application.add_handler(CommandHandler("education", self.education))
        application.add_handler(CommandHandler("help", self.help_command))
        
        # Callback handler'ı ekle
        application.add_handler(CallbackQueryHandler(self.button_callback))
        
        # Hata handler'ı ekle
        application.add_error_handler(self.error_handler)
        
        # Botu başlat
        logger.info("Bot başlatılıyor...")
        application.run_polling()
