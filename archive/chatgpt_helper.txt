from openai import OpenAI
import logging
from config import Config

class ChatGPTHelper:
    def __init__(self):
        self.config = Config()
        self.logger = logging.getLogger(__name__)
        self.client = OpenAI(
            base_url=self.config.OPENROUTER_BASE_URL,
            api_key=self.config.OPENROUTER_API_KEY
        )
        self.model = self.config.OPENROUTER_MODEL

    def get_ai_analysis(self, coin_name, technical_data, market_data, signals):
        try:
            technical_summary = self._format_technical_data(technical_data, signals)
            prompt = f"""
            Aşağıdaki {coin_name} kripto para analizini değerlendir ve Türkçe olarak yatırımcılar için anlaşılır bir yorum yap:

            **Piyasa Verileri:**
            - Güncel Fiyat: ${market_data['current_price']:,.2f}
            - 24s Değişim: {market_data['price_change_24h']:.2f}%
            - Piyasa Değeri: ${market_data['market_cap']:,.0f}
            - 24s Hacim: ${market_data['volume_24h']:,.0f}

            **Teknik Analiz:**
            {technical_summary}

            **Trading Sinyalleri:**
            - Genel Sinyal: {signals['overall_signal']}
            - Güven Oranı: {signals['confidence']}%

            Lütfen şu başlıklar altında analiz yap:
            1. **Genel Durum Değerlendirmesi** (2-3 cümle)
            2. **Teknik Göstergeler Yorumu** (RSI, MACD, MA'lar hakkında)
            3. **Kısa Vadeli Beklenti** (1-3 gün)
            4. **Orta Vadeli Beklenti** (1-2 hafta)
            5. **Risk Faktörleri** (dikkat edilmesi gerekenler)
            6. **Öneriler** (yatırımcılar için)

            Analizini profesyonel ama anlaşılır bir dille yaz. Emoji kullanma. Maksimum 300 kelime olsun.
            """
            messages = [
                {"role": "user", "content": prompt}
            ]
            response = self.client.chat.completions.create(
                model=self.model,
                messages=messages
            )
            return response.choices[0].message.content
        except Exception as e:
            self.logger.error(f"OpenRouter analizi hatası: {e}")
            return "❌ AI analizi alınırken hata oluştu."

    def get_market_sentiment(self, coin_list):
        try:
            prompt = f"""
            Aşağıdaki kripto paraların genel piyasa durumunu değerlendir:
            {', '.join(coin_list)}

            Şu başlıklar altında kısa bir değerlendirme yap:
            1. **Genel Piyasa Durumu** (yükseliş/düşüş trendi)
            2. **Öne Çıkan Kripto Paralar**
            3. **Risk Seviyesi**
            4. **Yatırımcı Tavsiyesi**

            Maksimum 200 kelime, Türkçe, emoji kullanma.
            """
            messages = [
                {"role": "user", "content": prompt}
            ]
            response = self.client.chat.completions.create(
                model=self.model,
                messages=messages
            )
            return response.choices[0].message.content
        except Exception as e:
            self.logger.error(f"OpenRouter piyasa duyarlılığı hatası: {e}")
            return "❌ Piyasa duyarlılığı analizi alınamadı."

    def get_educational_content(self, topic):
        try:
            topics = {
                'rsi': 'RSI (Relative Strength Index) göstergesi nedir ve nasıl kullanılır?',
                'macd': 'MACD göstergesi nedir ve trading sinyalleri nasıl okunur?',
                'bollinger': 'Bollinger Bands nedir ve nasıl yorumlanır?',
                'volume': 'Hacim analizi neden önemlidir?',
                'risk': 'Kripto para yatırımlarında risk yönetimi nasıl yapılır?',
                'technical': 'Teknik analiz temelleri nelerdir?'
            }
            if topic not in topics:
                return "❌ Bu konu hakkında eğitici içerik bulunamadı."
            prompt = f"""
            {topics[topic]} konusunda yeni başlayan yatırımcılar için Türkçe, anlaşılır bir açıklama yaz.

            Şu başlıklar altında organize et:
            1. **Tanım ve Açıklama**
            2. **Nasıl Hesaplanır/Okunur**
            3. **Trading Sinyalleri**
            4. **Pratik Örnekler**
            5. **Dikkat Edilmesi Gerekenler**

            Maksimum 400 kelime, emoji kullanma, teknik terimleri açıkla.
            """
            messages = [
                {"role": "user", "content": prompt}
            ]
            response = self.client.chat.completions.create(
                model=self.model,
                messages=messages
            )
            return response.choices[0].message.content
        except Exception as e:
            self.logger.error(f"OpenRouter eğitici içerik hatası: {e}")
            return "❌ Eğitici içerik oluşturulamadı."

    def get_news_analysis(self, news_headlines):
        try:
            if not news_headlines:
                return "Güncel haber bulunamadı."
            prompt = f"""
            Aşağıdaki kripto para haberlerini analiz et ve piyasaya etkisini değerlendir:

            {news_headlines}

            Şu başlıklar altında kısa analiz yap:
            1. **Haberlerin Genel Tonu** (olumlu/olumsuz/nötr)
            2. **Piyasaya Potansiyel Etkisi**
            3. **Yatırımcılar İçin Önemli Noktalar**

            Maksimum 250 kelime, Türkçe, emoji kullanma.
            """
            messages = [
                {"role": "user", "content": prompt}
            ]
            response = self.client.chat.completions.create(
                model=self.model,
                messages=messages
            )
            return response.choices[0].message.content
        except Exception as e:
            self.logger.error(f"OpenRouter haber analizi hatası: {e}")
            return "❌ Haber analizi yapılamadı."

    def _format_technical_data(self, technical_data, signals):
        try:
            current = technical_data.iloc[-1]
            summary = f"""
            - RSI: {current['rsi']:.1f} (Sinyal: {signals['rsi_signal']})
            - MACD: {current['macd']:.4f} (Sinyal: {signals['macd_signal']})
            - MA20: ${current['ma_20']:.2f}
            - MA50: ${current['ma_50']:.2f}
            - Bollinger Üst: ${current['bb_upper']:.2f}
            - Bollinger Alt: ${current['bb_lower']:.2f}
            - Hacim Oranı: {current['volume_ratio']:.2f}x
            """
            return summary
        except Exception as e:
            self.logger.error(f"Teknik veri formatlama hatası: {e}")
            return "Teknik veriler formatlanamadı."
